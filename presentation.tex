\section{Recherche Unimodale}
\begin{frame}{Méthodologie et Pipeline}
    \begin{columns}[T]
        \begin{column}{0.6\textwidth}
            \begin{itemize}
                \item \textbf{Dataset} : 10 000 images de voitures.
                \item \textbf{Indexation} : Extraction de descripteurs visuels.
                \item \textbf{Moteur} : Calcul de distances et Ranking.
            \end{itemize}
        \end{column}
        \begin{column}{0.4\textwidth}
            \centering \textit{[Insérer Interface Unimodale]}
        \end{column}
    \end{columns}
\end{frame}

\begin{frame}{Descripteurs Visuels}
    \begin{exampleblock}{Classiques}
        Couleur (HSV), Forme (HOG), Texture (LBP), Points (SIFT/ORB).
    \end{exampleblock}
    \begin{alertblock}{Deep Learning}
        ResNet50 (CNN) et ViT-B/16 (Vision Transformer).
    \end{alertblock}
\end{frame}

\begin{frame}{Résultats de l'évaluation}
    \begin{table}[]
        \centering
        \resizebox{0.8\textwidth}{!}{
        \begin{tabular}{lcccc}
            \toprule
            \textbf{Descripteur} & \textbf{P@50} & \textbf{mAP@50} & \textbf{R-Prec} \\ \midrule

            \bottomrule
        \end{tabular}

    \end{table}
    \centering Les modèles \textbf{Transformers} offrent les meilleures performances.
\end{frame}
