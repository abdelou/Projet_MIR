% --- Introduction ---
\section{Introduction}
\begin{frame}{Introduction et Objectifs}
    \begin{itemize}
        \item \textbf{Contexte} : Projet MIR (Multimedia Information Retrieval).
        \item \textbf{Objectif Principal} : Concevoir un moteur de recherche performant basé sur le contenu (CBIR).
        \item \textbf{Deux axes majeurs} :
        \begin{enumerate}
            \item \textbf{Recherche Unimodale} : Image-vers-Image sur une base de 10 000 voitures.
            \item \textbf{Recherche Multimodale} : Texte-vers-Image / Image-vers-Texte avec CLIP sur Flickr8K.
            \item \textbf{Déploiement} : Mise en ligne sur le Cloud (Render).
        \end{enumerate}
    \end{itemize}
\end{frame}

% --- Partie II : Multimodale ---
\section{Partie II : Moteur de Recherche Multimodal}

\begin{frame}{Partie II : CLIP et Flickr8K}
    \begin{itemize}
        \item \textbf{Motivation} : Dépasser la recherche par image pure $\rightarrow$ Requêtes Textuelles.
        \item \textbf{Dataset} : Flickr8K (8 000 images, 40 000 captions).
        \item \textbf{Modèle CLIP} :
        \begin{itemize}
            \item Espace latent commun (Image \& Texte).
            \item Projection dans une dimension 512.
            \item Pas de fine-tuning nécessaire (Zero-shot).
        \end{itemize}
    \end{itemize}
\end{frame}

\begin{frame}{Architecture Multimodale}
    \centering
    \begin{tikzpicture}[node distance=2cm, auto]
        \node (input) [draw, fill=blue!10] {Requête (Texte/Image)};
        \node (enc) [draw, right of=input, xshift=2cm, fill=green!10] {Encodeur CLIP};
        \node (faiss) [draw, right of=enc, xshift=2cm, fill=red!10] {Index FAISS};
        \node (results) [draw, below of=faiss, fill=yellow!10] {Top-$k$ Résultats};
        
        \draw[->] (input) -- (enc);
        \draw[->] (enc) -- (faiss);
        \draw[->] (faiss) -- (results);
    \end{tikzpicture}
    \vspace{0.5cm}
    \begin{itemize}
        \item Utilisation de \textbf{FAISS} pour une recherche de similarité ultra-rapide.
    \end{itemize}
\end{frame}

\begin{frame}{Résultats CLIP : Performance Globale}
    \begin{table}[]
        \centering
        \begin{tabular}{@{}lcc@{}}
            \toprule
            \textbf{Tâche} & \textbf{mAP} & \textbf{Recall@10} \\ \midrule
            \textbf{Texte $\rightarrow$ Image} & 0.4969         & 0.660             \\
            Image $\rightarrow$ Texte & 0.2979         & 0.533             \\ \bottomrule
        \end{tabular}
        \caption{Résumé des performances (Tableau 3.5)}

    \begin{itemize}
        \item Excellente capture de la sémantique pour les descriptions naturelles.
        \item Image-to-Text plus difficile (1 choix parmi 40 000).
    \end{itemize}
\end{frame}
